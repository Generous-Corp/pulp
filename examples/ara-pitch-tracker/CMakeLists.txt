# ara-pitch-tracker — the minimum ARA-aware CLAP plug-in.
#
# Demonstrates end-to-end integration: Processor overrides
# create_ara_document_controller(), CLAP adapter surfaces the factory
# via get_extension(kClapAraFactoryExtension). Real pitch analysis is
# left as an exercise — the point here is the host handshake.
#
# Workstream 06 slice A3.

# Only builds on platforms where CLAP is viable.
if(ANDROID OR IOS)
    return()
endif()

pulp_add_plugin(AraPitchTracker
    FORMATS          CLAP
    PLUGIN_NAME      "Pulp ARA Pitch Tracker"
    BUNDLE_ID        "com.pulp.examples.ara_pitch_tracker"
    MANUFACTURER     "Pulp"
    VERSION          "0.1.0"
    CATEGORY         Effect
    PLUGIN_CODE      "Pptr"
    MANUFACTURER_CODE "Pulp"
    SOURCES          src/pitch_tracker.cpp
)

# clap_entry.cpp pulls the Processor factory via PULP_CLAP_PLUGIN macro
# but the CLAP target is generated by pulp_add_plugin; attach our entry
# TU directly if it was created.
if(TARGET AraPitchTracker-CLAP)
    target_sources(AraPitchTracker-CLAP PRIVATE clap_entry.cpp)
    target_include_directories(AraPitchTracker-CLAP PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
endif()
